c=tk.StringVar()
d=tk.StringVar()
def btn1_clicked():
        switch_staus="on"
        tk.messagebox.showinfo(title='操作提示',message='蓝牙插座已开启')
        with Peripheral('C1:8F:CC:DC:57:19','random') as p:
            ch=p.getCharacteristics(uuid=TX_CHAR_UUID)[0]
            val=binascii.a2b_hex('AA0630306655')
            ch.write(val)
            p.disconnect()
